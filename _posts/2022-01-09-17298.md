# 백준 17298.cpp 오큰수 문제
## 문제
[![17298.cpp](/assets/images/algorithm/bj_17298/bj_17298.jpg)](https://www.acmicpc.net/problem/17298)
 <details>
 <summary>코드 자세히 보기</summary>

 ```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main(void)
{
 cin.tie(NULL);
 cout.tie(NULL);
 ios::sync_with_stdio(false);
 int n;
 cin >> n;
 vector<int> array(n, 0);
 vector<int> stack;
 vector<int> answer(n, 0);
 for (int i = 0; i < n; i++)
  cin >> array[i];
 answer[n - 1] = -1;
 for (int i = n - 1; i >= 0; i--)
 {
  bool flag = true;
  while (!stack.empty())
  {
   if (stack.back() > array[i])
   {
    flag = false;
    answer[i] = stack.back();
    break;
   }
   stack.pop_back();
  }
  if (flag == true)
   answer[i] = -1;
  stack.push_back(array[i]);
 }
 for (int j = 0; j < n; j++)
  cout << answer[j] << " ";
 cout << endl;
 return 0;
}
 ```
